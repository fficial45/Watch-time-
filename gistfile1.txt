    <script>
        let players = [];
        let isMuted = false;

        function loadVideos() {
            let url = document.getElementById("videoLink").value;
            let count = parseInt(document.getElementById("videoCount").value) || 1;
            let videoContainer = document.getElementById("videoContainer");
            let controls = document.getElementById("controls");

            videoContainer.innerHTML = "";
            players = [];

            let videoId = url.split("v=")[1]?.split("&")[0];
            if (!videoId) {
                alert("‚ùå ‡§ï‡•É‡§™‡§Ø‡§æ ‡§∏‡§π‡•Ä YouTube ‡§≤‡§ø‡§Ç‡§ï ‡§°‡§æ‡§≤‡•á‡§Ç!");
                return;
            }

            for (let i = 0; i < count; i++) {
                let div = document.createElement("div");
                div.innerHTML = `<div id="player${i}" class="video-frame"></div>`;
                videoContainer.appendChild(div);

                players.push(new YT.Player(`player${i}`, {
                    height: "160",
                    width: "280",
                    videoId: videoId,
                    playerVars: { 'rel': 0, 'showinfo': 0 },
                    events: {
                        'onReady': onPlayerReady
                    }
                }));
            }

            controls.style.display = "block"; // Controls ‡§ï‡•ã ‡§¶‡§ø‡§ñ‡§æ‡§è‡§Ç
        }

        function onPlayerReady(event) {
            let speed = parseFloat(document.getElementById("videoSpeed").value);
            event.target.setPlaybackRate(speed);
        }

        function playAllVideos() {
            players.forEach(player => {
                player.playVideo();
            });
        }

        function pauseAllVideos() {
            players.forEach(player => {
                player.pauseVideo();
            });
        }

        function toggleMute() {
            isMuted = !isMuted;
            players.forEach(player => {
                if (isMuted) {
                    player.mute();
                } else {
                    player.unMute();
                }
            });
        }
